2. Float 到 UInt 转换

为使 float 可以按位比较，需要进行特殊的位操作转换：

■ 符号位 (1 bit) ■ 指数位 (8 bits) ■ 尾数位 (23 bits)

Flip 技巧：可排序的 UInt

__device__ __forceinline__ unsigned int float_to_sortable_uint(float value) {
    unsigned int bits;
    memcpy(&bits, &value, sizeof(float));
    
    // Flip 技巧：使负数可以按 uint 排序
    if (bits & 0x80000000) {      // 负数：符号位为 1
        bits = ~bits;                  // 按位取反
    } else {                          // 正数：符号位为 0
        bits |= 0x80000000;            // 设置符号位为 1
    }
    return bits;
}

// 转换效果:
// 原始顺序: -∞ < -1.0 < 0 < 1.0 < +∞
// 转换后:    0  < 0x7F... < 0x80000000 < 0xFF... 
// 结果: 数值越大，uint 值越大！

3. 4-Pass 执行流程

每轮处理 8 位，从最高位到最低位逐步缩小 Top-K 所在范围：

1
Pass 1: 最高 8 位 (bits 31-24)
确定大致区间
输入所有 N 个元素，统计每个桶的分布。从最高桶向下累加，找到包含 Top-K 的桶。
2
Pass 2: 次高 8 位 (bits 23-16)
进一步细分
只处理 Pass 1 筛选出的桶中的元素，在该子集中继续统计分布。
3
Pass 3: 第 15-8 位
继续缩小范围
4
Pass 4: 最低 8 位 (bits 7-0)
精确定位
最终阈值确定后，收集所有 ≥ 阈值的元素。如果候选数 > K，使用传统算法选出 K 个。

5. 完整算法流程

1. 初始化阈值
threshold = 0, bit_start = 24，准备处理最高 8 位
2. 计算直方图
启动 radix_histogram_kernel，每个 Block 计算局部直方图
3. 合并直方图
merge_histograms_kernel 合并所有 Block 的直方图
4. 确定目标桶
从最高桶向下扫描，找到累计数 ≥ K 的桶
5. 更新阈值
bit_start -= 8，更新 threshold，进入下一轮
6. 重复 2-5 (共 4 轮)
直到处理完 32 位 (4 passes × 8 bits)
7. 收集候选元素
collect_candidates_kernel 收集所有 ≥ 最终阈值的元素
8. 最终 Top-K 选择
final_select_topk_kernel 从候选中选出精确的 K 个

7. 关键代码解析

7.1 直方图计算内核

template <int BLOCK_SIZE>
__global__ void radix_histogram_kernel(
    const float* input, int N,
    unsigned int* histograms,
    int bit_start,
    unsigned int current_threshold) 
{
    __shared__ unsigned int s_histogram[256];  // 256 桶

    int tid = threadIdx.x;
    int gid = tid + blockIdx.x * blockDim.x;
    int stride = blockDim.x * gridDim.x;

    // 初始化共享直方图
    for (int i = tid; i < 256; i += blockDim.x) {
        s_histogram[i] = 0;
    }
    __syncthreads();

    // 处理输入数据
    for (int idx = gid; idx < N; idx += stride) {
        float val = input[idx];
        unsigned int bits = float_to_sortable_uint(val);

        // 阈值检查：只处理高 bits 匹配的元素
        bool include = true;
        if (bit_start < 24) {
            unsigned int high_bits = bits >> (bit_start + 8);
            unsigned int threshold_high = current_threshold >> 8;
            include = (high_bits == threshold_high);
        }

        if (include) {
            // 提取当前 pass 的 8 位作为桶索引
            unsigned int key = (bits >> bit_start) & 0xFF;
            atomicAdd(&s_histogram[key], 1);  // 原子累加
        }
    }
    __syncthreads();

    // 写入全局内存
    unsigned int* block_hist = histograms + blockIdx.x * 256;
    for (int i = tid; i < 256; i += blockDim.x) {
        block_hist[i] = s_histogram[i];
    }
}

7.2 候选元素收集

__global__ void collect_candidates_kernel(
    const float* input, int N,
    TopKNode* candidates,
    int* candidate_count,
    unsigned int threshold_bits,
    int num_bits_checked) 
{
    int gid = threadIdx.x + blockIdx.x * blockDim.x;
    int stride = blockDim.x * gridDim.x;

    for (int idx = gid; idx < N; idx += stride) {
        float val = input[idx];
        unsigned int bits = float_to_sortable_uint(val);

        // 检查前 num_bits_checked 位是否匹配阈值
        unsigned int checked_bits = bits >> (32 - num_bits_checked);
        unsigned int threshold_checked = threshold_bits >> (32 - num_bits_checked);

        if (checked_bits == threshold_checked) {
            // 原子获取位置并写入候选数组
            int pos = atomicAdd(candidate_count, 1);
            if (pos < N) {
                candidates[pos].value = val;
                candidates[pos].index = idx;
            }
        }
    }
}

8. 总结

Radix Select 是工业级 Top-K 算法的核心，与 V1-V3 的思路完全不同：

无分支
直方图替代插入排序
避免分支预测失败
可扩展
支持大 K 值
不受寄存器限制
确定性
O(N × P) 复杂度
P 为常数 (4)
硬件友好
顺序扫描 + 原子操作
缓存友好
