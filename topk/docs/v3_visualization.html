2. V3 三阶段执行流程

1
Warp 级 Shuffle 归约
每个 Warp 32 线程使用 Shuffle 指令进行蝴蝶网络归约。 通过 5 步 (offset=16→8→4→2→1) 完成，最终 Lane 0 持有该 Warp 的 Top-K。 延迟极低：仅 5-10 周期！
2
Warp Leader 写入共享内存
每个 Warp 的 Lane 0 将结果写入共享内存。 相比 V2 的 256 次写入，V3 仅需 8 次 (假设 8 Warps)。 写入量减少 32 倍！
3
Warp 0 最终归约
只有 Warp 0 参与最终块级归约，节省其他 Warp 算力。 再次使用 Shuffle 指令合并 8 个 Warp 的结果。 Lane 0 写入全局内存。

5. 关键代码解析

5.1 Warp 归约函数

template <int K>
__device__ __forceinline__ void warp_reduce_topk(TopKNode* local_top) {
    // 锦标赛式归约：蝴蝶网络
    for (int offset = 16; offset > 0; offset /= 2) {
        // 从配对 lane 获取数据 (核心 Shuffle 指令!)
        for (int i = 0; i < K; ++i) {
            float partner_val = __shfl_down_sync(0xffffffff, local_top[i].value, offset);
            int partner_idx = __shfl_down_sync(0xffffffff, local_top[i].index, offset);

            // 合并逻辑: 如果 partner 有更好的值
            if (partner_val > local_top[K - 1].value) {
                int pos = K - 1;
                while (pos > 0 && partner_val > local_top[pos - 1].value) {
                    local_top[pos] = local_top[pos - 1];
                    pos--;
                }
                local_top[pos].value = partner_val;
                local_top[pos].index = partner_idx;
            }
        }
    }
    // 结果: Lane 0 持有该 Warp 的 Top-K
}

5.2 主内核核心逻辑

template <int K>
__global__ void topk_v3_warp_kernel(const float* input, int N, TopKNode* block_tops) {
    extern __shared__ TopKNode smem[];

    int tid = threadIdx.x;
    int lane_id = tid % 32;      // Warp 内的 lane
    int warp_id = tid / 32;      // 块内的 warp 索引
    int global_tid = tid + blockIdx.x * blockDim.x;

    // 1. 寄存器中维护局部 Top-K
    TopKNode local_top[K];
    // ... 初始化并扫描数据 ...

    // 2. Warp 级 Shuffle 归约 (核心优化!)
    warp_reduce_topk<K>(local_top);

    // 3. Warp leader (lane 0) 写入共享内存
    if (lane_id == 0) {
        for (int i = 0; i < K; ++i) {
            smem[warp_id * K + i] = local_top[i];
        }
    }
    __syncthreads();

    // 4. 只有 Warp 0 参与最终归约 (节省算力!)
    if (warp_id == 0) {
        // 从共享内存加载所有 Warp 结果
        TopKNode warp_top[K];
        if (lane_id < num_warps) {
            for (int i = 0; i < K; ++i) {
                warp_top[i] = smem[lane_id * K + i];
            }
        }

        // Warp 0 内再次 Shuffle 归约
        warp_reduce_topk<K>(warp_top);

        // Lane 0 写入全局内存
        if (lane_id == 0) {
            for (int i = 0; i < K; ++i) {
                block_tops[blockIdx.x * K + i] = warp_top[i];
            }
        }
    }
}
