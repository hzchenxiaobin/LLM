6. 关键代码解析

6.1 合并函数：merge_topk_in_smem

template <int K>
__device__ __forceinline__ void merge_topk_in_smem(
    TopKNode* smem, 
    int dst_offset, 
    int src_offset) 
{
    // 临时数组存储 2K 个元素
    TopKNode temp[2 * K];

    // 加载两个已排序的 Top-K 数组
    for (int i = 0; i < K; ++i) {
        temp[i] = smem[dst_offset + i];
        temp[i + K] = smem[src_offset + i];
    }

    // 选择排序取前 K 个 (K <= 256, 2K <= 512)
    for (int i = 0; i < K; ++i) {
        int max_idx = i;
        for (int j = i + 1; j < 2 * K; ++j) {
            if (temp[j].value > temp[max_idx].value) {
                max_idx = j;
            }
        }
        if (max_idx != i) {
            TopKNode swap = temp[i];
            temp[i] = temp[max_idx];
            temp[max_idx] = swap;
        }
    }

    // 写回目标位置
    for (int i = 0; i < K; ++i) {
        smem[dst_offset + i] = temp[i];
    }
}

6.2 主内核：topk_v2_block_kernel（简化版）

template <int K>
__global__ void topk_v2_block_kernel(
    const float* input, 
    int N, 
    TopKNode* block_tops) 
{
    extern __shared__ TopKNode smem[];  // 动态共享内存
    
    int tid = threadIdx.x;
    int block_id = blockIdx.x;

    // 1. 初始化局部 Top-K (寄存器)
    TopKNode local_top[K];
    for (int i = 0; i < K; ++i) {
        local_top[i].value = -1e20f;
        local_top[i].index = -1;
    }

    // 2. Grid-stride 循环扫描数据
    int elements_per_block = (N + gridDim.x - 1) / gridDim.x;
    int block_start = block_id * elements_per_block;
    int block_end = min(block_start + elements_per_block, N);

    for (int idx = block_start + tid; idx < block_end; idx += blockDim.x) {
        float val = input[idx];
        // 插入排序维护局部 Top-K...
    }

    // 3. 写入共享内存
    for (int i = 0; i < K; ++i) {
        smem[tid * K + i] = local_top[i];
    }
    __syncthreads();

    // 4. 树形归约
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            merge_topk_in_smem<K>(smem, tid * K, (tid + s) * K);
        }
        __syncthreads();
    }

    // 5. Thread 0 写入全局内存
    if (tid == 0) {
        for (int i = 0; i < K; ++i) {
            block_tops[block_id * K + i] = smem[i];
        }
    }
}

7. 总结

V2 版本的核心改进是利用共享内存进行块级数据合并，将全局内存写入量从 num_threads × K 减少到 num_blocks × K。在典型的 256 线程/Block 配置下，这是 256 倍的写入量减少。

学习要点

  • 内存层次意识：理解寄存器 → 共享内存 → L2 → 全局内存的速度差异
  • 归约模式：掌握树形归约（Tree Reduction）这一经典的并行算法模式
  • 同步机制：正确使用 __syncthreads() 避免数据竞争
  • 分治策略：Map-Reduce 思想在 GPU 并行计算中的应用

适用场景

  • ✅ 中小规模 K 值 (K <= 256)
  • ✅ 需要减少全局内存带宽压力的场景
  • ✅ 适合学习 GPU 内存层次和同步机制
