# V5: float4 向量化访问详解

## 概述

这是 CUDA 矩阵转置的**第五代也是最终优化版本**，通过 **`float4` 向量化内存访问** 实现极致性能，目标达到理论显存带宽的 **90%+**。

---

## 完整代码

```cpp
#include "transpose_common.h"

#define TILE_DIM    32
#define BLOCK_ROWS   8

// float4 向量化类型
struct float4_aligned {
    float x, y, z, w;
};

// 更简洁有效的 float4 向量化转置实现
// v5: float4 向量化访问
// 每次内存指令搬运16字节(4个float)，最大化总线利用率
// 目标：达到理论带宽的 90%+
__global__ void transpose_vectorized_kernel(const float *A, float *B, int N) {
    // 共享内存：TILE_DIM 行 x TILE_DIM+1 列，但以 float 存储
    __shared__ float tile[TILE_DIM][TILE_DIM + 1 + 4];  // 额外padding确保float4对齐

    int x = blockIdx.x * TILE_DIM + threadIdx.x;
    int y_base = blockIdx.y * TILE_DIM + threadIdx.y * 4;  // 每个线程处理4行

    // 以 float4 方式读取 A
    // 注意：这里需要确保内存对齐
    if (x < N / 4 && y_base < N) {
        // 读取4个连续的float4 (16个float)
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            int y = y_base + i * (TILE_DIM / 4);
            if (y < N) {
                float4_ALIGNED f4 = *((float4_ALIGNED *)&A[y * N + x * 4]);
                // 将float4的4个分量分散存储到共享内存的不同行
                tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 0] = f4.x;
                tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 1] = f4.y;
                tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 2] = f4.z;
                tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 3] = f4.w;
            }
        }
    }
    __syncthreads();

    // 转置后写入 B
    int x_base = blockIdx.y * TILE_DIM + threadIdx.x * 4;
    int y = blockIdx.x * TILE_DIM + threadIdx.y * 4;

    if (x_base < N && y < N) {
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            if (y + i * (TILE_DIM / 4) < N) {
                float4_ALIGNED f4;
                int row = threadIdx.x * 4;
                int col = threadIdx.y * 4 + i * (TILE_DIM / 4);
                f4.x = tile[col + 0][row + 0];
                f4.y = tile[col + 0][row + 1];
                f4.z = tile[col + 0][row + 2];
                f4.w = tile[col + 0][row + 3];
                *((float4_ALIGNED *)&B[(y + i * (TILE_DIM / 4)) * N + x_base]) = f4;
            }
        }
    }
}

void transpose_vectorized(const float *d_A, float *d_B, int N, cudaStream_t stream) {
    // 确保 N 是 32 的倍数
    dim3 block(TILE_DIM / 4, BLOCK_ROWS);  // 8 x 8 = 64 线程
    dim3 grid((N + TILE_DIM - 1) / TILE_DIM, (N + TILE_DIM - 1) / TILE_DIM);
    transpose_vectorized_kernel<<<grid, block, 0, stream>>>(d_A, d_B, N);
}
```

## 详细执行流程

### Step 1: float4 读取 A → 共享内存

```cpp
int x = blockIdx.x * TILE_DIM + threadIdx.x;           // 列 (0,4,8...)
int y_base = blockIdx.y * TILE_DIM + threadIdx.y * 4;  // 行起始

if (x < N / 4 && y_base < N) {
    #pragma unroll
    for (int i = 0; i < 4; i++) {
        int y = y_base + i * (TILE_DIM / 4);  // y + 0, y + 8, y + 16, y + 24
        if (y < N) {
            // 以 float4 方式读取 16 字节
            float4_ALIGNED f4 = *((float4_ALIGNED *)&A[y * N + x * 4]);
            
            // 解包到共享内存的不同行/列
            tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 0] = f4.x;
            tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 1] = f4.y;
            tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 2] = f4.z;
            tile[threadIdx.y * 4 + i][threadIdx.x * 4 + 3] = f4.w;
        }
    }
}
```

### Step 2: 坐标互换

```cpp
int x_base = blockIdx.y * TILE_DIM + threadIdx.x * 4;  // 转置后的列
```

**与 V4 相同，但注意 threadIdx.x/y 已经 ×4**

### Step 3: 打包 → float4 写入 B

```cpp
if (x_base < N && y < N) {
    #pragma unroll
    for (int i = 0; i < 4; i++) {
        if (y + i * (TILE_DIM / 4) < N) {
            float4_ALIGNED f4;
            int row = threadIdx.x * 4;
            int col = threadIdx.y * 4 + i * (TILE_DIM / 4);
            
            // 从共享内存打包到 float4
            f4.x = tile[col + 0][row + 0];
            f4.y = tile[col + 0][row + 1];
            f4.z = tile[col + 0][row + 2];
            f4.w = tile[col + 0][row + 3];
            
            // 向量化写入
            *((float4_ALIGNED *)&B[(y + i * (TILE_DIM / 4)) * N + x_base]) = f4;
        }
    }
}
```

### 1. 对齐要求

```cpp
// 额外的 padding 确保 float4 对齐
__shared__ float tile[TILE_DIM][TILE_DIM + 1 + 4];  // 32×37
//                              │      │   │
//                              │      │   └── 额外对齐 padding
//                              │      └───── Bank conflict padding
//                              └──────────── Tile 宽度
```

**为什么需要 +4？**
- `threadIdx.x * 4` 访问列 0, 4, 8, 12, 16, 20, 24, 28
- 写入 `tile[...][tx*4+0]` 到 `tile[...][tx*4+3]` (最多到列 31)
- 额外的 padding 确保编译器优化和边界安全

### 2. 类型转换技巧

```cpp
// 定义对齐的 float4 结构体
struct float4_aligned {
    float x, y, z, w;
};

// 读取：将 float* 转换为 float4_aligned*
float4_ALIGNED f4 = *((float4_ALIGNED *)&A[y * N + x * 4]);

// 写入：将 float4_aligned* 转换为 B 的地址
*((float4_ALIGNED *)&B[y * N + x]) = f4;
```

**注意：**
- 地址必须 16 字节对齐（`x * 4` 确保列对齐）
- 矩阵 N 应该是 32 的倍数以确保行对齐

### 3. 循环展开

```cpp
#pragma unroll
for (int i = 0; i < 4; i++) {
    // ...
}
```

**效果：**
- 编译器完全展开循环，消除循环开销
- 允许指令重排序优化
- 4 次迭代完全展开为顺序代码
